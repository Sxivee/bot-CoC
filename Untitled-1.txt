# Diccionario de traducciones
mensajes = {
    'es': {
        'saludo': 'Hola, bienvenido a la aplicación.',
        'despedida': 'Adiós, gracias por usar la aplicación.',
        'elige_idioma': 'Elige un idioma (es/en): ',
        'opcion_invalida': 'Opción no válida. Se usará español por defecto.'
    },
    'en': {
        'saludo': 'Hello, welcome to the application.',
        'despedida': 'Goodbye, thank you for using the application.',
        'elige_idioma': 'Choose a language (es/en): ',
        'opcion_invalida': 'Invalid option. Defaulting to Spanish.'
    }
}

# Elegir idioma
idioma = input(mensajes['es']['elige_idioma']).strip().lower()
if idioma not in mensajes:
    print(mensajes['en']['opcion_invalida'])
    idioma = 'es'

# Usar idioma elegido
print(mensajes[idioma]['saludo'])

print(mensajes[idioma]['despedida'])
